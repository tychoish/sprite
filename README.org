#+TITLE: sprite.el -- Manage subordinate Emacs daemon instances
#+OPTIONS: toc:2 num:nil

* Overview

=sprite.el= manages "sprite" Emacs daemons that are subordinate to a
parent Emacs instance.  Each sprite runs as a named =emacs --daemon=
subprocess and is addressable via =emacsclient=.

Sprite names follow the pattern =<parent>.<idx>.<unique-name>=, where
=idx= is the lowest integer not already in use for that parent.  The
registry persists across sessions via =savehist=, and the filesystem
state directory serves as the ground truth for discovery and recovery.

A companion module, =sprite-session.el=, provides idle-timer and
systemd-logind sleep hooks that application packages can register
against, decoupling trigger mechanisms from application-level responses.

Entry points:

- =sprite-list= -- open the sprite overview buffer
- =sprite-create= -- spawn a new sprite daemon
- =with-sprite= -- evaluate a form in a named sprite
- =sprite-get-next= -- get the next available sprite
- =sprite-get-or-create-next= -- get or create a sprite

* Installation
** package-vc.el

Requires Emacs 29.1+.  Run once to clone and register the package:

#+BEGIN_SRC emacs-lisp
(package-vc-install '(sprite :url "https://github.com/tychoish/sprite"))
#+END_SRC

** use-package with VC

Requires Emacs 30+.  Installs on first load if not already present:

#+BEGIN_SRC emacs-lisp
(use-package sprite
  :vc (:url "https://github.com/tychoish/sprite"))
#+END_SRC

To also load =sprite-session=:

#+BEGIN_SRC emacs-lisp
(use-package sprite-session
  :vc (:url "https://github.com/tychoish/sprite"))
#+END_SRC

** use-package with load-path

#+BEGIN_SRC sh
git clone https://github.com/tychoish/sprite ~/path/to/sprite
#+END_SRC

#+BEGIN_SRC emacs-lisp
(use-package sprite
  :load-path "~/path/to/sprite"
  :commands (sprite-list sprite-create sprite-get-next sprite-get-or-create-next))

(use-package sprite-session
  :load-path "~/path/to/sprite"
  :commands (sprite-session-start-idle-timer sprite-session-start-logind-watch))
#+END_SRC

    (defun sprite-doc--section-name (comment-text)
      (let* ((name (string-trim (car (split-string comment-text " -- "))))
             (name (replace-regexp-in-string "-" " " name)))
        (mapconcat (lambda (s) (if (string-match-p "\\." s) s (capitalize s)))
                   (split-string name "[ \t]+" t) " ")))

    (defun sprite-doc--format-doc (doc)
      (let* ((paras (split-string doc "\n\n" t))
             (fn-para (seq-find (lambda (p) (string-match "\\`(fn " (string-trim p))) paras))
             (rest-paras (seq-remove (lambda (p) (string-match "\\`(fn " (string-trim p))) paras)))
        (mapconcat #'identity
                   (delq nil
                         (list
                          (when fn-para (concat ": " (string-trim fn-para)))
                          (when rest-paras
                            (mapconcat
                             (lambda (para)
                               (if (string-match "\\`[ \t]" para)
                                   para
                                 (mapconcat #'string-trim (split-string para "\n" t) " ")))
                             rest-paras "\n\n"))))
                   "\n\n")))

    (defun sprite-doc--defkind (def-keyword)
      (cond
       ((member def-keyword '("defvar" "defconst" "defvar-local")) 'variable)
       ((equal def-keyword "defcustom") 'custom)
       (t 'function)))

    (defun sprite-doc--kind-tag (sym kind)
      (cond
       ((commandp sym) " [Command]")
       ((eq kind 'custom) " [Option]")
       ((eq kind 'variable) " [Variable]")
       (t "")))

    (defun sprite-doc--format-sym (name kind &optional heading)
      (let* ((sym (intern name))
             (fn-p (fboundp sym))
             (var-p (and (not fn-p) (boundp sym)))
             (doc (or (when fn-p (documentation sym))
                      (when var-p (documentation-property sym 'variable-documentation)))))
        (format "%s~%s~%s\n\n%s\n\n"
                (concat (or heading "*** ") (if doc "" "TODO "))
                name
                (sprite-doc--kind-tag sym kind)
                (if doc (sprite-doc--format-doc doc) "*no docstring*"))))

    (defun sprite-doc--include-p (name kind autoload-p)
      (and (not (string-match-p "[a-z]--" name))
           (cond
            ((eq kind 'variable) (memq 'variables include-kinds))
            ((eq kind 'custom)
             (or (memq 'customs include-kinds)
                 (and (memq scope '(exported autoloaded)) autoload-p)))
            (t
             (or (eq scope 'exported)
                 (and (eq scope 'autoloaded) autoload-p)
                 (eq scope 'interactive))))))

    (cond
     ;; Section-based collection for 'exported, 'autoloaded, and 'interactive.
     ;; Sections delimited by long ;;;;... ruler lines; the first ;; comment
     ;; after each ruler names the section.
     ((memq scope '(exported autoloaded interactive))
      (let (sections current-section current-syms after-ruler autoload-next)
        (seq-do
         (lambda (el-file)
           (with-temp-buffer
             (insert-file-contents el-file)
             (goto-char (point-min))
             (while (not (eobp))
               (let ((line (buffer-substring-no-properties
                            (line-beginning-position) (line-end-position))))
                 (cond
                  ((string-match "^;\\{10,\\}" line)
                   (when (and current-section current-syms)
                     (push (cons current-section (nreverse current-syms)) sections))
                   (setq current-section nil current-syms nil after-ruler t))
                  ((and after-ruler (string-match "^;; \\(.+\\)" line))
                   (setq current-section (sprite-doc--section-name (match-string 1 line))
                         after-ruler nil))
                  ((string-match "^;;;###autoload" line)
                   (setq autoload-next t))
                  ((string-match "^(\\(?:cl-\\)?\\(def[^ ]+\\) +'?\\([^ ()]+\\)" line)
                   (let* ((def-kw (match-string 1 line))
                          (sym-name (match-string 2 line))
                          (kind (sprite-doc--defkind def-kw)))
                     (when (and current-section
                                (sprite-doc--include-p sym-name kind autoload-next))
                       (push (list sym-name kind) current-syms))
                     (setq autoload-next nil)))
                  ((not (string-empty-p (string-trim line)))
                   (setq autoload-next nil))))
               (forward-line 1)))
           (when (and current-section current-syms)
             (push (cons current-section (nreverse current-syms)) sections))
           (setq current-section nil current-syms nil after-ruler nil autoload-next nil))
         el-files)
        (seq-do
         (lambda (section)
           (let ((syms (if (eq scope 'interactive)
                           (seq-filter (lambda (e) (commandp (intern (car e)))) (cdr section))
                         (cdr section))))
             (when syms
               (princ (format "** %s\n\n" (car section)))
               (seq-do (lambda (entry)
                         (princ (sprite-doc--format-sym (car entry) (cadr entry))))
                       syms))))
         (nreverse sections))))

     ;; Flat output for an explicit symbol list, in the given order.
     ((listp scope)
      (seq-do (lambda (name)
                (princ (sprite-doc--format-sym name 'function "** ")))
              scope))))
